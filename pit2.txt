                return;

            }

        }

        public class Ucenik
        {
            public string Ime { get; set; }
            public string Prezime { get; set; }
            public int[] Ocene { get; set; }

            public double Prosek()
            {
                return Ocene.Average();

        private void OsveziListView()
        {
            listView1.Items.Clear();

            double sumaProseka = 0;

            foreach (string linija in listBox1.Items)
            {
                string[] delovi = linija.Split('|');

                string ime = delovi[0];
                string prezime = delovi[1];

                int[] ocene = new int[6];
                for (int i = 0; i < 6; i++)
                {
                    ocene[i] = int.Parse(delovi[i + 2]);
                }

                double prosek = ocene.Average();
                sumaProseka += prosek;

                ListViewItem item = new ListViewItem(ime);
                item.SubItems.Add(prezime);

                foreach (int oc in ocene)
                {
                    item.SubItems.Add(oc.ToString());
                }

                item.SubItems.Add(prosek.ToString("0.00"));
                if (prosek >= 3.0)
                    item.BackColor = Color.LightGreen;
                else if (prosek < 3.0)
                    item.BackColor = Color.LightCoral;

                listView1.Items.Add(item);
            }

            if (listView1.Items.Count > 0)
            {
                double prosekOdeljenja = sumaProseka / listView1.Items.Count;
                label1.Text = "Prosek odeljenja: " + prosekOdeljenja.ToString("0.00");
            }

           
        }
